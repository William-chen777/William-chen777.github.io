将txt文档批量整合

发表于 2020-07-14 | 分类于 Python

批量读取txt文档整合到一个里面

import os
import re
filePath = 'C:\\Users\\ALIENWARE\\Desktop\\slide.window.dot-line.figure\\slide.window.dot-line.figure\\wt_for_cmt_ddm_drm_met\\wt_for_cmt_ddm_drm_met-rep3'      #在路径中要使用双斜杠
name_list = os.listdir(filePath)
up_list = []
mid_list = []
down_list = []                                  #定义三个列表

for i in name_list:
    if re.match('.*上.*', i) != None:
        up_list.append(i)

    elif re.match('.*中.*',i) !=None:
        mid_list.append(i)

    elif re.match('.*下.*', i) !=None:         #将txt文件按照上中下区别分开
        down_list.append(i)

#

name = []
for i in up_list:
    n_i = i[:25] + i[26:]
    name.append(n_i)                      #获取最终输出的文件名

#

for m in range(0,18):
    filename = name[m]

    f1 = open(up_list[m], 'r')
    f2 = open(mid_list[m], 'r')
    f3 = open(down_list[m], 'r')

    contents1 = f1.read()
    contents2 = f2.read()
    contents3 = f3.read()

    with open('C:\\Users\\ALIENWARE\\Desktop\\slide.window.dot-line.figure\\slide.window.dot-line.figure\\wt_for_cmt_ddm_drm_met\\wt_for_cmt_ddm_drm_met-rep3\\txt\\'+filename, "a") as f:
        f.write(contents1)
        f.write(contents2)
        f.write(contents3)
